function numberOfPairs(gloves) {
    let glove = [...gloves].sort((a, b) => a.localeCompare(b));
    let pairs = 0;
    for (let i = 0; i < glove.length; i++) {
        if (glove[i - 1] === glove[i]) {
            pairs++;
            i++;
        }
    }
    return pairs;
}
